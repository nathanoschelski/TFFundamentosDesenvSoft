package com.sales.entidades;
import com.sales.casosdeuso.*;

        this.nome = nome;
    }

        return nome;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

    public List<VendaDTO> getVendas() {
        return vendas;
    }

    public void setVendas(List<VendaDTO> vendas) {
        this.vendas = vendas;
    }


}
